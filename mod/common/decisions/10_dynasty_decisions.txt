create_cadet_branch_decision = {
	picture = "gfx/interface/illustrations/decisions/decision_dynasty_house.dds"

	# TODO: Dynamic description based on tiers

	major = yes
	ai_check_interval = 60

	is_shown = {
		is_landed = yes
		exists = house
		exists = house.house_head
		is_house_head = no
		primary_title.tier > tier_barony
		NOR = {
			government_has_flag = government_is_holy_order
			government_has_flag = government_is_theocracy
			AND = {
				faith = { has_doctrine = doctrine_theocracy_temporal }
				has_council_position = councillor_court_chaplain
			}
		}
	}

	is_valid = {
		government_allows = create_cadet_branches

		house.house_head = {
			custom_description = {
				text = create_cadet_branch_decision_succession_line
				subject = root
				NOT = {
					any_held_title = {
						place_in_line_of_succession = {
							target = root
							value <= 3
						}
					}
				}
			}
		}
		trigger_if = {
			limit = { government_has_flag = government_is_clan }

			trigger_if = {
				limit = { is_independent_ruler = yes }
				is_independent_ruler = yes
			}
			trigger_else_if = {
				limit = {
					NOT = {
						any_liege_or_above = {
							house = root.house
						}
					}
				}
				custom_description = {
					text = create_cadet_branch_decision_no_liege_house
				}
				create_cadet_branch_decision_no_liege_house_trigger = yes
			}
			trigger_else = {
				custom_description = {
					text = create_cadet_branch_decision_liege_house
				}
				create_cadet_branch_decision_liege_house_trigger = yes
			}
		}

		custom_description = {
			text = create_cadet_branch_decision_ancestor_in_house
			NOT = {
				any_ancestor = {
					house = root.house
				}
			}
		}
		
		trigger_if = {
			limit = {
				is_married = yes
				is_male = yes
			}
			patrilinear_marriage = yes
		}
		trigger_if = {
			limit = {
				is_married = yes
				is_female = yes
			}
			matrilinear_marriage = yes
		}
		trigger_if = { #Males of female-dominated faith must already be patrilineally married before taking this decision.
			limit = {
				is_married = no
				is_female = no
				faith = {
					has_doctrine = doctrine_gender_female_dominated
				}
			}
			patrilinear_marriage = yes
		}
		trigger_if = { #Females of male-dominated faith must already be matrilineally married before taking this decision.
			limit = {
				is_married = no
				is_female = yes
				faith = {
					has_doctrine = doctrine_gender_male_dominated
				}
			}
			matrilinear_marriage = yes
		}
		trigger_if = {
			limit = {
				has_trait = devoted
			}
			NOT = {
				has_trait = devoted
			}
		}
		trigger_if = {
			limit = {
				has_trait = bastard
			}
			NOT = {
				has_trait = bastard
			}
		}
	}

	is_valid_showing_failures_only = {
		is_available_adult_or_is_commanding = yes # From is_available_adult
		is_landed = yes
	}

	effect = {
		# NOT CLAN
		if = {
			limit = {
				NOT = { government_has_flag = government_is_clan }
			}
			found_cadet_house_decision_effect = {
				CHARACTER = root
				PRESTIGE = major_prestige_gain
			}
		}
		else = {
			# CLAN AND INDEPENDENT
			if = {
				limit = {
					is_independent_ruler = yes
				}
				create_cadet_branch_decision_independent_effect = yes
			}
			# CLAN AND NOT INDEPENDENT AND NO LIEGE IS PART OF HOUSE
			else_if = {
				limit = {
					NOT = {
						any_liege_or_above = {
							house = root.house
						}
					}
				}
				create_cadet_branch_decision_no_liege_house_effect = yes
			}
			# CLAN AND NOT INDEPENDENT AND ANY LIEGE IS PART OF HOUSE
			else = {
				create_cadet_branch_decision_liege_house_effect = yes
			}
		}

		remove_trait = decadent
		remove_trait = extolled
	}
	
	ai_potential = {
		always = yes
	}

	ai_will_do = {
		base = 30
		
		modifier = {
			add = 30
			highest_held_title_tier = tier_duchy
		}

		modifier = {
			add = 40
			highest_held_title_tier >= tier_kingdom
		}

		# NOT independent
		modifier = {
			add = -30 # No vassal count allowed to create cadet branch
			is_independent_ruler = no
		}

		modifier = {
			add = -20 # -50 -> -20
			# NOT Clan
			NOT = { government_has_flag = government_is_clan }
			is_independent_ruler = no
			any_liege_or_above = {
				house = root.house # same dynasty -> same house
			}
		}

		modifier = {
			factor = 0
			any_child = {
				is_alive = yes
				count < 3
			}
		}
		
		modifier = {
			factor = 0
			any_player = {
				OR = {
					any_child = {
						OR = {
							is_child_of = root
							is_grandchild_of = root
							is_great_grandchild_of = root
						}
					}
					any_child = {
						even_if_dead = yes
						any_child = {
							OR = {
								is_child_of = root
								is_grandchild_of = root
								is_great_grandchild_of = root
							}
						}
					}
					any_child = {
						even_if_dead = yes
						any_child = {
							even_if_dead = yes
							any_child = {
								OR = {
									is_child_of = root
									is_grandchild_of = root
									is_great_grandchild_of = root
								}
							}
						}
					}
				}
			}
		}

		# House landed rulers size
		#   5 |-> 0.5
		#  10 |-> 1.0
		modifier = {
			factor = {
				value = number_of_landed_house_members_value
				multiply = 0.1
			}
			house = {
				any_house_member = { 
					count <= 10 
					is_landed = yes
				}
			}
		}

		#   15 |-> 1.25
		#   20 |-> 1.5
		#   25 |-> 1.75
		# >=30 |-> 2.0
		modifier = {
			factor = {
				value = number_of_landed_house_members_value
				subtract = 10
				multiply = 0.05
				add = 1
				max = 2
			}
			house = {
				any_house_member = { 
					count > 10 
					is_landed = yes
				}
			}
		}

		# Clan
		# c = (ai_boldness - dread) * 1/100
		# factor = -1/50 * opinion + c
		# c = -1, op = -50  => f = 0
		# c = -1, op = -100 => f = 1
		# c = 0,  op = 0    => f = 0
		# c = 0,  op = -50  => f = 1
		# c = 1,  op = 50   => f = 0
		# c = 1,  op = 0    => f = 1
		modifier = {
			factor = {
				value = "opinion(house.house_head)"
				multiply = -0.02
				add = {
					value = ai_boldness
					subtract = house.house_head.dread
					multiply = 0.01
				}
				min = 0
				max = 1
			}
			government_has_flag = government_is_clan
		}

		# House Unity - Antagonistic
		modifier = {
			factor = 1.2
			government_has_flag = government_is_clan
			house = {
				has_house_unity_stage = antagonistic
			}
		}

		# House Unity - Harmonious
		modifier = {
			factor = 0.8
			government_has_flag = government_is_clan
			house = {
				has_house_unity_stage = harmonious
			}
		}

		# Clan house head is weak
		modifier = {
			factor = 1.25
			government_has_flag = government_is_clan
			house.house_head = {
				OR = {
					is_adult = no
					is_imprisoned = yes
					has_contagious_deadly_disease_trigger = yes
					is_incapable = yes
				}
			}
		}

		# Patriarch/Matriarch trait
		modifier = {
			factor = 1.2
			has_trait = family_first
		}

		# Ambitious
		modifier = {
			factor = 1.1
			has_trait = ambitious
		}

		# Accused of Decadence
		modifier = {
			factor = 1.1
			has_trait = decadent
		}
	}
}
